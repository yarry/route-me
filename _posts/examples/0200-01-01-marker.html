--- date: 0200-01-06 category: examples layout: examples title: Drawing a Mapbox marker rotation: vertical description: Creating a default marker with a callout and title. tags: - marker - point - pin - callout - popup --- @implementation MyViewController - (void)viewDidLoad { [super viewDidLoad]; [[RMConfiguration sharedInstance] setAccessToken:@""]; RMMapboxSource *tileSource = [[RMMapboxSource alloc] initWithMapID:@"mapbox.emerald"]; RMMapView *mapView = [[RMMapView alloc] initWithFrame:self.view.bounds andTilesource:tileSource]; [self.view addSubview:mapView]; // add callout title RMPointAnnotation *annotation = [[RMPointAnnotation alloc] initWithMapView:mapView coordinate:mapView.centerCoordinate andTitle:@"Hello, world!"]; [mapView addAnnotation:annotation]; }
