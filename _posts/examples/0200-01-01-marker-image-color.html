--- date: 0200-01-07 category: examples layout: examples title: Custom icon and color on Mapbox marker description: Creating a Mapbox marker with a custom icon and color. notes: This example uses a Maki icon. tags: - marker - point - pin - callout - popup - maki - icon - color - image --- @implementation MyViewController - (void)viewDidLoad { [super viewDidLoad]; [[RMConfiguration sharedInstance] setAccessToken:@""]; RMMapboxSource *tileSource = [[RMMapboxSource alloc] initWithMapID:@"mapbox.emerald"]; RMMapView *mapView = [[RMMapView alloc] initWithFrame:self.view.bounds andTilesource:tileSource]; // you will need to add a to the ViewController.h file mapView.delegate = self; [self.view addSubview:mapView]; // add callout title RMAnnotation *annotation = [[RMAnnotation alloc] initWithMapView:mapView coordinate:mapView.centerCoordinate andTitle:@"Mapbox Garage"]; [mapView addAnnotation:annotation]; } - (RMMapLayer *)mapView:(RMMapView *)mapView layerForAnnotation:(RMAnnotation *)annotation { if (annotation.isUserLocationAnnotation) return nil; // add Maki icon and color the marker RMMarker *marker = [[RMMarker alloc] initWithMapboxMarkerImage:@"rocket" tintColor: [UIColor colorWithRed:0.224 green:0.671 blue:0.780 alpha:1.000]]; marker.canShowCallout = YES; return marker; }
